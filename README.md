## Структура проекта

```bash
src/
├── components/         # UI-компоненты
│   ├── Character/      # Персонаж и его состояния
│   ├── Dialogue/       # Компонент диалогов
│   ├── Games/          # Мини-игры
│   └── UI/             # Кнопки, иконки и т.д.
├── pages/              # Страницы приложения
│   ├── Home/           # Главная страница
│   ├── Game/           # Игровой процесс
│   └── Results/        # Результаты и достижения
├── services/           # API и сервисы
│   ├── AudioService.ts # Управление звуком
│   └── GameService.ts  # Логика игры
├── data/               # Данные игры
│   ├── scenes/         # Сцены с заданиями
│   └── letters.ts      # Информация о буквах
├── utils/              # Утилиты
├── hooks/              # Пользовательские хуки
├── types/              # TypeScript типы
└── styles/             # Глобальные стили
```
